//Swap Numbers
import java.util.*;
public class Main
{
	public static void main(String[] args) {
		int a, b, temp;
		Scanner sc = new Scanner(System.in);
		a = sc.nextInt();
		b = sc.nextInt();
		System.out.println("The numbers are " + a + " " + b);
		temp = a;
		a=b;
		b=temp;
		System.out.println("Swapped numbers " + a + " " + b);
	}
}

//Second Largest Element
import java.util.*;
public class Main
{
	public static void main(String[] args) {
	    Scanner sc = new Scanner(System.in);
	    int n;
	    System.out.println("Give the size of the array: ");
	    n = sc.nextInt();
	    int arr[] = new int[n];
	    for(int i=0;i<n;i++)
	    {
	        arr[i]=sc.nextInt();
	    }
	    Arrays.sort(arr);
	    System.out.println("Second largest element is " + arr[n-2]);
	}
}

//Number of digits 
import java.util.*;
public class Main
{
	public static void main(String[] args) {
	    Scanner sc= new Scanner(System.in);
	    int n;
	    int rev;
	    int count=0;
	    n = sc.nextInt();
	    while(n>0)
	    {
	      rev = n%10;
	      count++;
	      n=n/10;
	    }
	    System.out.println(count);
	}
}

//reverse of array
import java.util.*;
public class Main
{
	public static void main(String[] args) {
		Scanner sc = new Scanner(System.in);
		int n;
	    System.out.print("Enter the size: ");
	    n = sc.nextInt();
		int a[] = new int[n];
		for(int i=0;i<n;i++)
		{
		    a[i] = sc.nextInt();
		}
		for(int i=n-1;i>=0;i--)
		{
		    System.out.print(a[i] + " ");
		}
		
	}
}
